#### Update: `src/lib/api.ts` - Add error handling for RLS

```typescript
import { createClient } from '@supabase/supabase-js';

const supabase = createClient(
  import.meta.env.VITE_SUPABASE_URL,
  import.meta.env.VITE_SUPABASE_ANON_KEY
);

export const sbFetch = async (
  method: 'GET' | 'POST' | 'PATCH' | 'DELETE',
  table: string,
  data?: any,
  filters?: Record<string, any>
): Promise<any> => {
  try {
    let query = supabase.from(table);

    if (method === 'GET') {
      Object.entries(filters || {}).forEach(([key, value]) => {
        query = query.eq(key, value);
      });
      const { data: result, error } = await query;
      if (error) {
        if (error.code === 'PGRST301') {
          console.warn(`RLS policy denied access to ${table}`);
        }
        throw error;
      }
      return result;
    }

    if (method === 'POST') {
      const { data: result, error } = await query.insert([data]);
      if (error) {
        if (error.code === 'PGRST301') {
          console.warn(`RLS policy denied INSERT on ${table}`);
        }
        throw error;
      }
      return result;
    }

    if (method === 'PATCH') {
      Object.entries(filters || {}).forEach(([key, value]) => {
        query = query.eq(key, value);
      });
      const { data: result, error } = await query.update(data);
      if (error) {
        if (error.code === 'PGRST301') {
          console.warn(`RLS policy denied UPDATE on ${table}`);
        }
        throw error;
      }
      return result;
    }

    if (method === 'DELETE') {
      Object.entries(filters || {}).forEach(([key, value]) => {
        query = query.eq(key, value);
      });
      const { error } = await query.delete();
      if (error) {
        if (error.code === 'PGRST301') {
          console.warn(`RLS policy denied DELETE on ${table}`);
        }
        throw error;
      }
    }
  } catch (err) {
    logger.error(`sbFetch error: ${method} ${table}`, err);
    throw err;
  }
};
```

## 🔒 SECURITY & RLS BEST PRACTICES

### Policy Architecture

All tables follow this RLS pattern:

```sql
-- Read policy (SELECT)
CREATE POLICY "Users see data"
ON table_name
FOR SELECT
USING (true);  -- Adjust: USE auth.uid() for real auth

-- Insert policy (INSERT)
CREATE POLICY "Users insert data"
ON table_name
FOR INSERT
WITH CHECK (true);  -- Adjust: USE auth.uid() for real auth

-- Update policy (UPDATE)
CREATE POLICY "Users update data"
ON table_name
FOR UPDATE
USING (true)    -- Adjust: USE auth.uid() for real auth
WITH CHECK (true);

-- Delete policy (DELETE)
CREATE POLICY "Users delete data"
ON table_name
FOR DELETE
USING (true);   -- Adjust: USE auth.uid() for real auth
```

### Migration Path for Authentication (P3.3.1 - Future Phase)

When integrating real authentication (Supabase Auth, OAuth, etc.):

CREATE TABLE users (

CREATE TABLE obra_owners (

  user_id UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
  obra_id UUID NOT NULL REFERENCES obras(id) ON DELETE CASCADE,
  role TEXT DEFAULT 'owner', -- 'owner', 'supervisor', 'viewer'
  UNIQUE(user_id, obra_id)

CREATE POLICY "Users see own obras"
ON obras
FOR SELECT
USING (
  id IN (
    SELECT obra_id FROM obra_owners

  )
);

## 🐛 TROUBLESHOOTING

### Issue: "RLS policy denied access"
**Solution:**
1. Verify RLS is enabled: `SELECT * FROM pg_tables WHERE rowsecurity = true`
2. Check policies exist: `SELECT * FROM pg_policies`
3. Temporarily disable RLS for debugging:
   ```sql
   ALTER TABLE servicos DISABLE ROW LEVEL SECURITY;
   ```
4. Re-enable after testing:
   ```sql
   ALTER TABLE servicos ENABLE ROW LEVEL SECURITY;
   ```

### Issue: "Constraint violation on INSERT"
**Solution:**
1. Check data values match constraints:
   - `avanco_atual` must be 0-100
   - `data_fim` must be >= `data_inicio`
   - `status_atual` must be one of: nao_iniciado, em_andamento, concluido
2. Example query to validate before INSERT:
   ```sql
   SELECT 
     CASE WHEN avanco_atual NOT BETWEEN 0 AND 100 THEN 'Invalid avanço' END,
     CASE WHEN data_fim < data_inicio THEN 'Invalid date range' END
   FROM servicos;
   ```

### Issue: "Slow queries after index creation"
**Solution:**
1. Indexes need to be analyzed:
   ```sql
   ANALYZE servicos;
   ANALYZE diario_obra;
   ANALYZE pendencias;
   ```
2. Check query plans:
   ```sql
   EXPLAIN ANALYZE
   SELECT * FROM servicos 
   WHERE obra_id = 'uuid' 
     AND data_inicio BETWEEN '2026-04-01' AND '2026-05-01'
   ORDER BY data_inicio;
   ```
